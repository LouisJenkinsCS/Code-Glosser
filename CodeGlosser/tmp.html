package springbreak;

 * @author theif519

public class SpringBreak {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        System.out.println(firstDigit(123));
        System.out.println(lastDigit(123));
        System.out.println(digits(123));
        System.out.println(numInWords("Mary had a little lamb"));
        System.out.println(vowels("How many vowels are there in the alphabet?"));
    }
    public static int firstDigit(int n)
    {
        String step1 = Integer.toString(n);
        char step2 = step1.charAt(0);
        return Character.getNumericValue(step2);
    }
    public static int lastDigit(int n)
    {
        String step1 = Integer.toString(n);
        char step2 = step1.charAt(step1.length() - 1);
        return Character.getNumericValue(step2);
    }
    public static int digits(int n)
    {
        String step1 = Integer.toString(n);
        return step1.length();
    }
    public static int numInWords(String str)
    {
        int blank = 0;
        char[] step1 = str.toCharArray();
        for(int i = 0; i < str.length() - 1; i++)
        {
          //if (Character.isWhitespace(step1[i])) 
            if(step1[i] == ' ')
            {
                blank++;
            }
        }
        blank++;
        return blank;
    }
    public static int vowels(String str)
    {
        int vowels = 0;
        str.toLowerCase();
        char[] step1 = str.toCharArray();
        for(int i = 0; i < str.length() - 1; i++)
        {
            if (step1[i] == 'a' || step1[i] == 'e' || step1[i] == 'i' || step1[i] == 'o' || step1[i] == 'u') 
            {
                vowels++;
            }
        }
        return vowels;
    }
}
