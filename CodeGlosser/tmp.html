package LanguageTemplate;

import java.util.stream.Stream;

/**
 *
 * @author Louis
 * 
 * Exception thrown when there are not enough tokens for desired input. Used to
 * give the user a more friendly and helpful error message. Note that calls to
 * the superclass constructor must be called first, but we don't have the 
 * appropriate string until after processing all tokens. We fix this issue by
 * conducting a proper stream that using the MapReduce pattern
 */
public class NotEnoughTokens extends RuntimeException {

    public NotEnoughTokens(Token... expected) {
        super(
                Stream.of(expected)
                .map(Token::toString)
                .map(StringBuilder::new)
                .peek((b) -> b.insert(0,"\t").append("\n"))
                .reduce(new StringBuilder("Expected Tokens: {\n"), StringBuilder::append)
                .append("}")
                .toString()
        );
    }
}
